QUANTIDADE_DE_ALUNOS = 3

ENQUANTO i < QUANTIDADE_DE_ALUNOS FAÇA
  LEIA ALUNOS[i].NOME
  LEIA ALUNOS[i].MATRICULA
  LEIA ALUNOS[i].NOTA1
  LEIA ALUNOS[i].NOTA2
FIM-ENQUANTO

ENQUANTO i < QUANTIDADE_DE_ALUNOS FAÇA
  ESCREVA ALUNOS[i].MATRICULA
  ESCREVA ALUNOS[i].NOME
  ESCREVA (ALUNOS[i].NOTA1 + ALUNOS[i].NOTA2)/2
FIM-ENQUANTO
